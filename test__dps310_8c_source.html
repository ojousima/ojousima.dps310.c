60 {
61  switch(reg)

64  TEST_ASSERT(0U == (value & (~DPS310_PRES_CFG_WMASK)));
65  break;
66  case DPS310_TEMP_CFG_REG:
67  TEST_ASSERT(0U == (value & (~DPS310_TEMP_CFG_WMASK)));
68  break;
69  case DPS310_MEAS_CFG_REG:
70  TEST_ASSERT(0U == (value & (~DPS310_MEAS_CFG_WMASK)));
71  break;
72  case DPS310_CFG_REG:
73  TEST_ASSERT(0U == (value & (~DPS310_CFG_WMASK)));
74  break;
75  case DPS310_RST_CFG_REG:
76  TEST_ASSERT(0U == (value & (~DPS310_RST_CFG_WMASK)));
77  break;
78  case DPS310_EFUSE_0_REG:
79  TEST_ASSERT((DPS310_EFUSE_0_VAL == value) || (0 == value));
80  efuse_writes++;
81  break;
82  case DPS310_EFUSE_1_REG:
83  TEST_ASSERT((DPS310_EFUSE_1_VAL == value) || (0 == value));
84  efuse_writes++;
85  break;
86  case DPS310_EFUSE_2_REG:
87  TEST_ASSERT((DPS310_EFUSE_2_VAL == value));
88  efuse_writes++;
89  break;
90  default:
91  TEST_ASSERT(0U);
92  break;
93  }

127 static void simulate_measurement_action(const uint8_t data)
128 {
129  switch(data)

137  dps310_registers[DPS310_TEMP_VAL_REG + 2] = 0x00U;

143 static void simulate_write_action(const uint8_t reg_addr, const uint8_t data)
144 {
145  switch(reg_addr)
146  {

150  init_dps_regs();
151  drdy_ms += DPS310_COEF_DELAY_MS;
152  soft_resets++;
153  }
154  break;
155  case DPS310_MEAS_CFG_REG:
156  simulate_measurement_action(data);
157  break;
158 
159  default:
160  break;
161  }
