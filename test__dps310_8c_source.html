79 static void assert_on_write_error(const uint8_t reg, const uint8_t value)
80 {
81  switch(reg)

83  case DPS310_PRES_CFG_REG:
84  TEST_ASSERT(0U == (value & (~DPS310_PRES_CFG_WMASK)));
85  break;
86  case DPS310_TEMP_CFG_REG:
87  TEST_ASSERT(0U == (value & (~DPS310_TEMP_CFG_WMASK)));
88  break;
89  case DPS310_MEAS_CFG_REG:
90  TEST_ASSERT(0U == (value & (~DPS310_MEAS_CFG_WMASK)));
91  break;
92  case DPS310_CFG_REG:
93  TEST_ASSERT(0U == (value & (~DPS310_CFG_WMASK)));
94  break;
95  case DPS310_RST_CFG_REG:
96  TEST_ASSERT(0U == (value & (~DPS310_RST_CFG_WMASK)));
97  break;
98  case DPS310_EFUSE_0_REG:
99  TEST_ASSERT((DPS310_EFUSE_0_VAL == value) || (0 == value));
100  efuse_writes++;
101  break;
102  case DPS310_EFUSE_1_REG:
103  TEST_ASSERT((DPS310_EFUSE_1_VAL == value) || (0 == value));
104  efuse_writes++;
105  break;
106  case DPS310_EFUSE_2_REG:
107  TEST_ASSERT((DPS310_EFUSE_2_VAL == value));
108  efuse_writes++;
109  break;
110  default:
111  TEST_ASSERT(0U);
112  break;
113  }

194  // For OS 1 register value 0x7A1200 and C0 -100, C1 1 => -34.7412 C
195  // scaled_raw 15.2587
196  dps310_registers[DPS310_TEMP_VAL_REG] = 0x7AU;
197  dps310_registers[DPS310_TEMP_VAL_REG + 1] = 0x12U;
198  dps310_registers[DPS310_TEMP_VAL_REG + 2] = 0x00U;
199  dps310_registers[DPS310_MEAS_CFG_REG] &= ~DPS310_MEAS_CFG_WMASK;
200  break;

206  dps310_registers[DPS310_PRES_VAL_REG + 1] = 0x12U;

211  default:
212  break;

216 static void simulate_write_action(const uint8_t reg_addr, const uint8_t data)
217 {
218  switch(reg_addr)
