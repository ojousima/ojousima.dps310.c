21 static int32_t twos_complement (const uint32_t value, const uint8_t bits)
22 {
23  const int32_t max_signed_value_exclusive = (int32_t) (1U << (bits - 1U));
24  int32_t complement = (int32_t) value;
25 
26  if (complement >= max_signed_value_exclusive)
27  {
28  complement -= 2 * max_signed_value_exclusive;
29  }
30 
31  return complement;
32 }

149 static dps310_status_t read_coefs (dps310_ctx_t * const ctx)
150 {
151  uint8_t coefs[DPS310_COEF_REG_LEN] = {0};
152  dps310_status_t reg_read_ret = ctx->read (ctx->comm_ctx, DPS310_COEF_START_REG,
153  coefs, DPS310_COEF_REG_LEN);
154  uint32_t reg_val = (((uint32_t) coefs[DPS310_COEF_C0_OFFSET] * 256U)
155  + ((uint32_t) coefs[DPS310_COEF_C0_OFFSET + 1U]))
156  >> DPS310_COEF_C0_SHIFT;
157  ctx->c0 = twos_complement (reg_val, DPS310_COEF_C0_LEN_BITS);
158  reg_val = (((uint32_t) coefs[DPS310_COEF_C1_OFFSET] * 256U)
159  + ((uint32_t) coefs[DPS310_COEF_C1_OFFSET + 1U]))
160  & ((1U << DPS310_COEF_C1_LEN_BITS) - 1U);
161  ctx->c1 = twos_complement (reg_val, DPS310_COEF_C1_LEN_BITS);
162  reg_val = (((uint32_t) coefs[DPS310_COEF_C00_OFFSET] * 256U * 256U)
163  + ((uint32_t) coefs[DPS310_COEF_C00_OFFSET + 1U] * 256U)
164  + ((uint32_t) coefs[DPS310_COEF_C00_OFFSET + 2U]))
165  >> DPS310_COEF_C00_SHIFT;
166  ctx->c00 = twos_complement (reg_val, DPS310_COEF_C00_LEN_BITS);
167  reg_val = (((uint32_t) coefs[DPS310_COEF_C10_OFFSET] * 256U * 256U)
168  + ((uint32_t) coefs[DPS310_COEF_C10_OFFSET + 1U] * 256U)
169  + ((uint32_t) coefs[DPS310_COEF_C10_OFFSET + 2U]))
170  & ((1U << DPS310_COEF_C10_LEN_BITS) - 1U);
171  ctx->c10 = twos_complement (reg_val, DPS310_COEF_C10_LEN_BITS);
172  reg_val = (((uint32_t) coefs[DPS310_COEF_C01_OFFSET] * 256U)
173  + ((uint32_t) coefs[DPS310_COEF_C01_OFFSET + 1U]));
174  ctx->c01 = twos_complement (reg_val, DPS310_COEF_C01_LEN_BITS);
175  reg_val = (((uint32_t) coefs[DPS310_COEF_C11_OFFSET] * 256U)
176  + ((uint32_t) coefs[DPS310_COEF_C11_OFFSET + 1U]));
177  ctx->c11 = twos_complement (reg_val, DPS310_COEF_C01_LEN_BITS);
178  reg_val = (((uint32_t) coefs[DPS310_COEF_C20_OFFSET] * 256U)
179  + ((uint32_t) coefs[DPS310_COEF_C20_OFFSET + 1U]));
180  ctx->c20 = twos_complement (reg_val, DPS310_COEF_C20_LEN_BITS);
181  reg_val = (((uint32_t) coefs[DPS310_COEF_C21_OFFSET] * 256U)
182  + ((uint32_t) coefs[DPS310_COEF_C21_OFFSET + 1U]));
183  ctx->c21 = twos_complement (reg_val, DPS310_COEF_C21_LEN_BITS);

185  + ((uint32_t) coefs[DPS310_COEF_C30_OFFSET + 1U]));

187  return reg_read_ret;

244 static const dps310_status_t efuse_write (dps310_ctx_t * const ctx)
245 {
246  dps310_status_t ret_code = ctx_ready_check (ctx);
247 
248  if (DPS310_SUCCESS == ret_code)
249  {

251  {

706 static float calculate_pressure (const dps310_ctx_t * const ctx, const int32_t raw)
707 {
708  uint32_t sf = os_to_scale_factor (ctx->pres_osr);
709  float Praw_sc = ((float) raw) / ((float) sf);
710  float Traw_sc = ctx->last_temp_scal;
711  return ctx->c00
712  + Praw_sc * ((float) ctx->c10 + Praw_sc * ((float) ctx->c20 + Praw_sc *
713  (float) ctx->c30))
714  + Traw_sc * (float) ctx->c01 + Traw_sc * Praw_sc * ((float) ctx->c11 + Praw_sc *
715  (float) ctx->c21);
716 }

719 regs_to_i32 (const uint8_t * const regs)
720 {
721  uint32_t b24_value = (regs[0U] << 16U)
722  + (regs[1U] << 8U)
723  + regs[2U];
724  return twos_complement (b24_value, 24U);
725 }

874 dps310_status_t dps310_get_cont_results (dps310_ctx_t * const ctx, float * const temp,
875  uint8_t * const temp_count, float * const pres, uint8_t * const pres_count)
876 {
877  dps310_status_t err_code = ctx_cont_check (ctx);
878  uint8_t temp_entries = 0U;
879  uint8_t pres_entries = 0U;
880 
881  if (DPS310_SUCCESS == err_code)
882  {
883  uint8_t fifo_entry[3U] = {0};
884  bool output_full = false;
885 
886  while ((!fifo_entry_is_eof (fifo_entry)) && (!output_full)
887  && (DPS310_SUCCESS == err_code))
888  {
889  err_code |= ctx->read (ctx->comm_ctx, DPS310_FIFO_VAL_REG, fifo_entry,
890  DPS310_FIFO_VAL_LEN);
891 
892  if (fifo_entry_is_pres (fifo_entry))
893  {
894  float result = calculate_pressure (ctx, regs_to_i32 (fifo_entry));
895  pres[pres_entries++] = result;
896  output_full = (pres_entries == *pres_count);
897  }
898  else if (!fifo_entry_is_eof (fifo_entry))
899  {
900  float result = calculate_temperature (ctx, regs_to_i32 (fifo_entry));
901  temp[temp_entries++] = result;
902  output_full = (temp_entries == *temp_count);
903  }
904  else
905  {
906  // EOF
907  *temp_count = temp_entries;
908  *pres_count = pres_entries;
909  }
910 
911  if (DPS310_SUCCESS != err_code)
912  {
913  err_code |= DPS310_BUS_ERROR;
914  ctx->device_status = DPS310_BUS_ERROR;
915  }
916  }
917  }
918 
919  return err_code;
920 }
